45

48 NC = parser.get_ncols();
49 leftovers = NC % chunk_size;
50 nchunks = atleastone(NC / chunk_size + (leftovers != 0));

62 tatami::cast_Index_to_container_size<std::vector<Category> >(NR); // check that resize is safe.

70 auto chunk = (c - 1) / chunk_size;
71 --r;

73 ++num_per_chunk[chunk][r];

80 });

82 parser.scan_real([&](Index_ r, Index_ c, double val) -> void {
83 handler(r, c, categorize(val));
84 });
85 } else {

87 }

89 allocate_rows(
90 max_per_chunk,
91 num_per_chunk,
92 identities8,
93 identities16,
94 identities32,
95 store8,
96 store16,
97 store32,
98 assigned_category,
99 assigned_position
100 );
101 }
102
103 // Now allocating.
104 {
105 std::vector<std::vector<std::size_t> > output_positions(nchunks);
106 tatami::cast_Index_to_container_size<decltype(output_positions.front())>(NR); // check that resize is safe.
107 for (decltype(nchunks) chunk = 0; chunk < nchunks; ++chunk) {
108 output_positions[chunk].resize(NR);
109 for (decltype(NR) r = 0; r < NR; ++r) {
110 output_positions[chunk][r] = get_sparse_ptr(store8, store16, store32, assigned_category, assigned_position, chunk, r);
111 }
112 }
113
114 auto reader = create();
115 byteme::PerByteSerial<char, decltype(&reader)> pb(&reader);
116 eminem::Parser<decltype(&pb), Index_> parser(&pb, eopt);
117
118 auto handler = [&](Index_ r, Index_ c, auto val) -> void {
119 --c;
120 Index_ chunk = c / chunk_size;
121 Index_ offset = c % chunk_size;
122 --r;
123 fill_sparse_value(store8, store16, store32, assigned_category[chunk][r], chunk, offset, val, output_positions[chunk][r]++);
124 };
125
126 parser.scan_preamble();
127 const auto& banner = parser.get_banner();
128 if (banner.field == eminem::Field::INTEGER) {
129 parser.template scan_integer<std::uint32_t>([&](Index_ r, Index_ c, std::uint32_t val) -> void {
130 handler(r, c, val);
131 });
132 } else if (banner.field == eminem::Field::DOUBLE || banner.field == eminem::Field::REAL) {
133 parser.scan_real([&](Index_ r, Index_ c, double val) -> void {
134 handler(r, c, val);
135 });
136 }
137
138 // Checking that the column indices are sorted properly.
139 auto sorter = [&](auto& store) -> void {
140 std::vector<std::pair<typename decltype(store[0].index)::value_type, typename decltype(store[0].value)::value_type> > buffer;
141 buffer.reserve(chunk_size);
142
143 for (auto& st : store) {
144 auto num_ptr = st.ptr.size();
145 for (decltype(num_ptr) r = 1; r < num_ptr; ++r) {
146 auto start = st.ptr[r - 1], end = st.ptr[r];
147
148 if (!std::is_sorted(st.index.begin() + start, st.index.begin() + end)) {
149 buffer.clear();
150 for (auto i = start; i < end; ++i) {
151 buffer.emplace_back(st.index[i], st.value[i]);
152 }
153
154 std::sort(buffer.begin(), buffer.end());
155 auto bIt = buffer.begin();
156 for (auto i = start; i < end; ++i, ++bIt) {
157 st.index[i] = bIt->first;
158 st.value[i] = bIt->second;
159 }
160 }
161 }
162 }
163 };
164
165 sorter(store8);
166 sorter(store16);
167 sorter(store32);
168 }
169
170 return consolidate_matrices<Value_, Index_>(
171 identities8,
172 identities16,
173 identities32,
174 std::move(store8),
175 std::move(store16),
176 std::move(store32),
177 NR,
178 chunk_size,
179 leftovers
180 );
